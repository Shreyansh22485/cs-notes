```cpp
    //     int n = nums.size();

    long long maxAlternatingSum(vector<int>& nums) {
        int n = nums.size();
        // vector<vector<long long>> dp(n,vector<long long>(2+1,-1));
        // return f(0,1,nums,dp);

        vector<vector<long long>> dp(n+1,vector<long long>(2,0)); // 1 baesd indexing
        // dp[i][0] is on adding the ith index element the length is even

        for(int i= 1;i<=n;i++){ //       1based indexsing
            dp[i][0] = max(dp[i-1][1] - nums[i-1] ,dp[i-1][0]);
            dp[i][1] = max(dp[i-1][0] + nums[i-1] ,dp[i-1][1]);
        }

        return max(dp[n][0],dp[n][1]);


    }
};
```
- Time Complexity: O(n) - We are iterating through the array of length n once, performing constant time operations in each iteration
- Space Complexity: O(n) - We are using a 2D dp array of size (n+1) x 2 to store the results of subproblems, which requires O(n) space.
