```cpp
class Solution {
public:
    const int mod = 1e9 + 7;
    int f(int ind ,int maxSoFar , int n , int m , int k, vector<vector<vector<int>>>& dp){
        if(ind == n){
            if(k==0) return 1;
            return 0;
        }
        if(k==-1) return 0;
        if(dp[ind][maxSoFar+1][k] != -1) return dp[ind][maxSoFar+1][k];


        int result=0;
        for(int i = 1 ; i<=m;i++){
            if(maxSoFar < i){
                result = (result%mod  + f(ind+1,i,n,m,k-1,dp) % mod) % mod;
            }
            else{
                result =  (result%mod  + f(ind+1,maxSoFar,n,m,k,dp) %mod ) % mod;
            }
        }
        return dp[ind][maxSoFar+1][k] = result % mod;
    }
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<int>>> dp(n+1,vector<vector<int>>(m+2,vector<int>(k+1,-1)));
        return f(0,-1,n,m,k,dp);
    }
};
```
- Time Complexity: O(n * m * k) - We are using a 3D dp array of size (n+1) x (m+2) x (k+1) to store the results of subproblems, and we are iterating through the array of length n, performing constant time operations in each iteration.
- Space Complexity: O(n * m * k) - We are using a 3D dp array of size (n+1) x (m+2) x (k+1) to store the results of subproblems, which requires O(n * m * k) space.
