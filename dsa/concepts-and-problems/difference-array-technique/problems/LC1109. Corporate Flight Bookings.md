```cpp
class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int>diff(n);
        for(auto& booking: bookings){
            int first = booking[0]-1;
            int last = booking[1]-1;
            int seats = booking[2];

            diff[first] += seats;
            if(last+1 < n)
                diff[last+1] -= seats;

        }

        vector<int> ans(n);
        int totalSum = 0;
        ans[0] = diff[0];
        for(int i=1;i<n;i++){
            diff[i] += diff[i-1];
            ans[i]=diff[i];
        }
        return ans;
    }
};
```
- Time Complexity : O(N + M) where N is the number of bookings and M is the range of the numbers (100 in this case).
- Space Complexity : O(M) for the difference array.
