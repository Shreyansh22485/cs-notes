```cpp
class Solution {
public:
    const int MOD = 1e9 + 7;
    int maxSumRangeQuery(vector<int>& nums, vector<vector<int>>& requests) {
        int n = nums.size();
        vector<int> diff(n);

        for(auto& req: requests){
            int start = req[0];
            int end = req[1];

            diff[start] += 1;
            if(end+1 < n)
                diff[end+1] -= 1;
        }

        for(int i=1;i<n;i++){
            diff[i] += diff[i-1];
        }

        sort(diff.begin(),diff.end(),greater<>());
        sort(nums.begin(),nums.end(),greater<>());

        int totalSum = 0;

        for(int i=0;i<n;i++){
            totalSum = (totalSum + (diff[i]*1ll*nums[i])) % MOD;
        }
        return totalSum;
    }
};
```
- Time Complexity : O(N log N) where N is the number of requests.
- Space Complexity : O(N) where N is the number of requests.
