```cpp
class Solution {
public:
    bool isZeroArray(vector<int>& nums, vector<vector<int>>& queries) {
        int n = nums.size();
        vector<int> changes(n,0);
        for(auto& query : queries){
            int l = query[0];
            int r = query[1];

            changes[l] += -1;
            if(r+1<n) changes[r+1] -= -1;
        }
        for(int i=1;i<n;i++){
            changes[i] = changes[i] + changes[i-1];
        }

        for(int i=0;i<n;i++){
            if(nums[i] + changes[i] > 0) return false;
        }
        return true;
    }
};
```
- Time Complexity: O(n+q)
- Space Complexity: O(n)
