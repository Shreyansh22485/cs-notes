```cpp
class Solution {
public:
    string shiftingLetters(string s, vector<vector<int>>& shifts) {
        int n = s.size();
        vector<int> change(n,0);
        for(auto& shift: shifts){
            int start = shift[0];
            int end = shift[1];
            int direction = shift[2];

            change[start] += (direction == 1)? 1:-1;
            if(end + 1 < n)
            change[end+1] -= (direction == 1)? 1:-1;
        }
        int cumSum = 0;
        for(int i=0;i<n;i++){
            cumSum += change[i];
            int shift = (cumSum % 26);
            if(shift < 0) shift+=26;
            s[i] = ((s[i]-'a') + shift)%26 + 'a';
        }
        return s;
    }
};
```
- Time Complexity: O(n)
- Space Complexity: O(n)
