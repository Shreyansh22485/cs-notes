```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        if(!list1 && !list2) return NULL;
        if(list1 && !list2) return list1;
        if(list2 && !list1) return list2;
        ListNode* head=NULL;
        ListNode* start=NULL;
        ListNode* temp1=list1;
        ListNode* temp2=list2;
        if(list1->val <=list2->val){
            start=list1;
            temp1=temp1->next;
        }
        else{
            start=list2;
            temp2=temp2->next;

        }
        head=start;
        while(temp1 && temp2){
            if(temp1->val<temp2->val){
                start->next=temp1;
                temp1=temp1->next;

            }
            else{
                start->next=temp2;
                temp2=temp2->next;
            }
            start=start->next;
        }
        if(temp1){
            start->next=temp1;
        }
        if( temp2){
            start->next=temp2;
        }
        return head;
    }
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        // ListNode* l1=lists[0];

        //     mergedNode=mergeTwoLists(mergedNode,lists[i]);
        // }
        // return mergedNode;

        //Optimal
        priority_queue<pair<int,ListNode*>,vector<pair<int,ListNode*>>,greater<pair<int,ListNode*>>> pq;
        for(int i=0;i<lists.size();i++){
            if(lists[i])
            pq.push({lists[i]->val,lists[i]});
        }
        ListNode* dummy=new ListNode(-1);
        ListNode* temp = dummy;
        while(!pq.empty()){
            ListNode* node=pq.top().second;
            pq.pop();
            temp->next=node;
            if(node->next) pq.push({node->next->val,node->next});
            temp=temp->next;
            

        }
        return dummy->next;
    }
};
```
- Time Complexity: O(N log K) where N is the total number of nodes in all lists and K is the number of linked lists. Each node is pushed and popped from the priority queue once, and each operation takes O(log K) time.
- Space Complexity: O(K) for the priority queue, which stores at most K nodes at any time.
