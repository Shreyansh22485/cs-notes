```cpp
class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        
        map<int,int> mpp;
        for(auto& trip : trips){
            mpp[trip[1]] += trip[0];
            mpp[trip[2]] -= trip[0];
        }

        int cnt = 0;
        for(auto it = mpp.begin() ; it != mpp.end(); it++){
            cnt += it->second;
            if(cnt > capacity){
                return false;
            }
        }
        return true;
    }
};
```
- **Time Complexity** - O(N*logN) due to sorting.
- **Space Complexity** - O(N) for storing the events.
