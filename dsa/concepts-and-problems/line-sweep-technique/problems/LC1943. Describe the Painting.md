```cpp
class Solution {
public:
    vector<vector<long long>> splitPainting(vector<vector<int>>& segments) {
        map<int,long long> mp;
        for(auto& segment: segments){
            int start = segment[0];
            int end = segment[1];
            int color = segment[2];

            mp[start] += color;
            mp[end] -= color;
        }

        long long cumSum = 0;
        vector<vector<long long>> ans;
        for(auto it=mp.begin();next(it)!=mp.end();it++){
            cumSum += it->second;
            it -> second = cumSum;
            int left = it->first;
            int right = next(it)->first;
            long long mix = it->second;
            if(mix > 0){
                ans.push_back({left,right,mix});    
            }
        }
        return ans;

    }
};
```
- Time Complexity : O(N log N) where N is the number of segments.
- Space Complexity : O(N) where N is the number of segments.
