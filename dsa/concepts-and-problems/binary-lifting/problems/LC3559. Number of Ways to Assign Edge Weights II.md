# Intuition
ans = **2^(d-1)** where d is the number of edges in the path from u to v.

$$
d = depth[u] + depth[v] - 2*depth[lca(u,v)]
$$

```cpp
class Solution {
public:
    const int MOD = 1e9 + 7;
    int rows;
    int cols;

    vector<int> depth;
    unordered_map<int,vector<int>> adj;
    vector<vector<int>> ancestorTable; 

    void dfs(int node, int parent){
        ancestorTable[node][0] = parent;

        for(auto& adjNode: adj[node]){
            if(adjNode != parent){
                depth[adjNode] = depth[node] + 1;
                dfs(adjNode,node);
            }
        }
    }

    void buildAncestorTable(){
        for(int j=1;j<cols;j++){
            for(int node=0;node<rows;node++){
                if(ancestorTable[node][j-1] != -1)
                    ancestorTable[node][j] = ancestorTable[ancestorTable[node][j-1]][j-1];
            }
        }
    }

    int lca(int u, int v){
        if(depth[u] < depth[v]){
            return lca(v,u);
        }

        int diff = depth[u] - depth[v];
        for(int j=0;j<cols;j++){
            if(diff & (1<<j)){
                u = ancestorTable[u][j];
            }
        }

        if(u == v)return u;

        for(int j=cols-1;j>=0;j--){
            if(ancestorTable[u][j] == -1) continue;
            if(ancestorTable[u][j] != ancestorTable[v][j]){
                u = ancestorTable[u][j];
                v = ancestorTable[v][j];
            }
        }

        return ancestorTable[u][0];
    }

    vector<int> assignEdgeWeights(vector<vector<int>>& edges, vector<vector<int>>& queries) {
       for (auto& e : edges) {
            adj[e[0]-1].push_back(e[1]-1);
            adj[e[1]-1].push_back(e[0]-1);

        }
        int n = adj.size();
        rows = n;
        cols = log2(n) + 1;
        depth.resize(n,0);
        ancestorTable.resize(rows,vector<int>(cols,-1));
        dfs(0,-1);
        buildAncestorTable();
        vector<int> pow2(n+1);
        pow2[0] = 1;
        for(int i=1;i<=n;i++){
            pow2[i] = (2LL * pow2[i-1]) % MOD ;
        }
        vector<int> results;
        for(auto& q: queries){
            int u = q[0]-1;
            int v = q[1]-1;

            int d = depth[u] + depth[v] - 2*depth[lca(u,v)];
            if(d == 0)
                results.push_back(0);
            else
                results.push_back(pow2[d-1]);
        }
        return results;
    
    }
};
```
- Time Complexity: O(n log n) for preprocessing the ancestorTable and O(q log n) for answering q queries, where n is the number of nodes and q is the number of queries.
- Space Complexity: O(n log n) for the ancestorTable and O(n) for the depth array, resulting in O(n log n) overall.
