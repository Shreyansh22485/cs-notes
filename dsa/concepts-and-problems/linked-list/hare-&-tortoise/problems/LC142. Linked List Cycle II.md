```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* slow=head;
        ListNode* fast=head;
        while(fast && fast->next){
            slow=slow->next;
            fast=fast->next->next;
            if(slow==fast){
                slow=head;
                int length=0;
                while(slow!=fast){
                    slow=slow->next;
                    fast=fast->next;
                    length++;
                }
                cout<<"Length of the cycle is: "<<length<<endl;
                return slow;
            }
        }
        return NULL;
        
    }
};
```
- Time Complexity: O(n), where n is the number of nodes in the linked list. The algorithm traverses the linked list at most twice.
- Space Complexity: O(1), as we are using only a constant amount of extra space for the two pointers (slow and fast). The input linked list is modified in place, but we are not using any additional data structures that grow with the input size.
