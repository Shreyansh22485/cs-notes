```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        if(head==NULL || head->next==NULL) return head;
        ListNode* fast=head;
        ListNode* slow=head;
        while(fast && fast->next){
            fast=fast->next->next;
            slow=slow->next;
        }
        return slow; // slow is the middle node in case of odd number of nodes and the second middle node in case of even number of nodes

        // if we want to return the first middle node in case of even number of nodes, we can use the following code:
        // while(fast && fast->next && fast->next->next){
        //     fast=fast->next->next;
        //     slow=slow->next;
        // }
        // return slow;
    }
};
```
- Time Complexity: O(n), where n is the number of nodes in the linked list. The algorithm traverses the linked list at most once.
- Space Complexity: O(1), as we are using only a constant amount of extra space for the two pointers (slow and fast). The input linked list is modified in place, but we are not using any additional data structures that grow with the input size.
