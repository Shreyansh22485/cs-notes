```cpp
class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        if(grid[0][0] == 1 || grid[n-1][m-1] == 1) return -1;
        queue<pair<int,int>>q;
        q.push({0,0});
        // vector<vector<int>> vis(n,vector<int>(m,0)); 
        // vis[0][0] = 1;

        grid[0][0] = 1; // We can use grid as vis array
        int dx[] = {-1,0,1,0,-1,1,1,-1};
        int dy[] = {0,1,0,-1,-1,1,-1,1};
        int level = 0;

        while(!q.empty()){
            int size = q.size();
            while(size--){
                auto it = q.front();
                q.pop();
                int i = it.first;
                int j = it.second;
            
                
                if(i == n-1 && j == m-1) return level+1;

                for(int k =0;k<8;k++){
                    int r = i + dx[k];
                    int c = j + dy[k];

                    if(r>=0 && r<n && c>=0 && c< m && grid[r][c] == 0){
                        grid[r][c] = 1;
                        q.push({r,c});
                    }
                }

            }
            level++;
        }

        return -1;
    }
};
```
- Time Complexity: O(n*m) where n and m are the dimensions of the grid.
- Space Complexity: O(n*m) for the distance array and the queue in the worst case.
