```cpp
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m=grid.size();
        int n=grid[0].size();
        queue<pair<int,int>> q;
        vector<vector<int>> vis(m,vector<int>(n,0));
        int total=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==2){
                    q.push({i,j});
                    vis[i][j]=2;
                }
                else{
                    if(grid[i][j]==1)
                    total++;
                }
            }
        }
        int cnt=0;
        int time=0;
        int dx[]={-1,0,1,0};
        int dy[]={0,1,0,-1};
        while(!q.empty()){
            int size = q.size();
            while(size--){
                auto it=q.front();
                q.pop();
                int row=it.first;
                int col=it.second;

                for(int k=0;k<4;k++){
                    int nrow=row+dx[k];
                    int ncol=col+dy[k];

                    if(nrow>=0 && nrow<m && ncol>=0 && ncol<n && vis[nrow][ncol]==0 && grid[nrow][ncol]==1){
                        q.push({nrow,ncol});
                        vis[nrow][ncol]=2;
                        cnt++;
                    }
                }
            }
            if(!q.empty()) time++;
            
        }
        if(cnt!=total) return -1;
        return time;
        
    }
};
```
- Time Complexity: O(m*n), where m is the number of rows and n is the number of columns in the grid. We traverse each cell at most once.
- Space Complexity: O(m*n) for the queue and the visited array.
