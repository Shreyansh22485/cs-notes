# Intution - Why Euler Path

- There is a Starting point and an Ending point in the graph. 
- Visiting all the edges exactly once

```cpp
class Solution {
public:
    void findEulerPath(int u, unordered_map<int,vector<int>>& adj, vector<int>& eulerPath){
        while(!adj[u].empty()){
            int v = adj[u].back();
            adj[u].pop_back();
            findEulerPath(v,adj,eulerPath);
        }
        eulerPath.push_back(u);
    }
    vector<vector<int>> validArrangement(vector<vector<int>>& pairs) {
        unordered_map<int,vector<int>> adj;
        unordered_map<int,int> indegree;
        unordered_map<int,int> outdegree;
        for(auto& pair: pairs){
            int u = pair[0];
            int v = pair[1];

            adj[u].push_back(v);
            indegree[v]++;
            outdegree[u]++;
        }

        int start = pairs[0][0]; // The Euler path exist but if no starting node , then it is euler circuit , therefore any node is starting node
        for(auto& it : adj){
            int node = it.first;
            int out = outdegree[node];
            int in = indegree[node];
            cout << in << " " << out << endl;
            if(out - in == 1){
                start = node;
                break;
            }
        }
        
        vector<int> eulerPath;
        findEulerPath(start,adj,eulerPath);
        reverse(eulerPath.begin(),eulerPath.end());

        vector<vector<int>> res;
        for(int i=0;i<eulerPath.size()-1;i++){
            res.push_back({eulerPath[i],eulerPath[i+1]});
        }
        return res;
    }
};
```
- Time Complexity: O(V + E), where V is the number of vertices and E is the number of edges in the graph.
- Space Complexity: O(V + E) for storing the adjacency list and O(V) for the in-degree and out-degree arrays.
