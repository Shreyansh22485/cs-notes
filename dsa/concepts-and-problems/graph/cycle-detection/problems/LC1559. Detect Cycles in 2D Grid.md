```cpp
class Solution {
public:

    bool dfs(int i, int j, int pi, int pj,vector<vector<bool>>& vis,vector<vector<char>>& grid ){
        vis[i][j] = true;
        int n = grid.size();
        int m = grid[0].size();
        int dx[] = {-1,0,1,0};
        int dy[] = {0,1,0,-1};

        for(int k=0;k<4;k++){
            int r = i + dx[k];
            int c = j + dy[k];

            if(r>=0 && r< n && c>=0 && c<m && grid[r][c] == grid[i][j]){
                if(!vis[r][c]){
                    if(dfs(r,c,i,j,vis,grid) == true) return true;
                }
                else if(r != pi && c != pj ){
                    return true;
                }
            }

        }
        return false;
    }

    bool containsCycle(vector<vector<char>>& grid) {
        int n = grid.size();
        int m = grid[0].size();

        vector<vector<bool>> vis(n,vector<bool>(m,false));

        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(!vis[i][j]){
                    if(dfs(i,j,-1,-1,vis,grid) == true){
                        return true;
                    }
                }
            }

        }
        return false;
    }
};
```
- Space Complexity: O(n*m) for the visited array and recursion stack.
- Time Complexity: O(n*m) for traversing all cells in the grid.
