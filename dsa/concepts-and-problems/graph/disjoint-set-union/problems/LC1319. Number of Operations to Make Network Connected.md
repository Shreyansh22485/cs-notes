```cpp
class Disjoint{
    private:
    vector<int> parent;
    vector<int> size;
    vector<int> rank;

    public:
    Disjoint(int n){
        rank.resize(n+1,0);
        size.resize(n+1,1);
        parent.resize(n+1);
        for(int i=0;i<n+1;i++){
            parent[i]=i;
        }
    }
    int findUlParent(int node){
        if(parent[node]==node) return node;
        return parent[node]=findUlParent(parent[node]);
    }
    void unionByRank(int u,int v){
        int ulp_u=findUlParent(u);
        int ulp_v=findUlParent(v);
        if(ulp_u==ulp_v) return;
        if(rank[ulp_u] < rank[ulp_v]){
            parent[ulp_u]=ulp_v;
        }
        else if(rank[ulp_v] < rank[ulp_u]){
            parent[ulp_v]=ulp_u;
        }
        else{
            parent[ulp_v]=ulp_u;
            rank[ulp_u]++;
        }

    }
    void unionBySize(int u, int v){
        int ulp_u=findUlParent(u);
        int ulp_v=findUlParent(v);
        if(ulp_u==ulp_v) return;
        if(size[ulp_u] < size[ulp_v]){
            parent[ulp_u]=ulp_v;
            size[ulp_v]+=size[ulp_u];
        }
        else{
            parent[ulp_v]=ulp_u;
            size[ulp_u]+=size[ulp_v];
        }
    }

};
class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        int cntExtra=0;
        Disjoint ds(n);
        for(auto it:connections){
            int u=it[0];
            int v=it[1];
            if(ds.findUlParent(u)==ds.findUlParent(v)){
                cntExtra++;
            }
            else{
                ds.unionBySize(u,v);
            }
        }
        int cntC=0;
        for(int i=0;i<n;i++){
            if(ds.findUlParent(i)==i){
                cntC++;
            }
        }
        if(cntExtra >= cntC-1) return cntC-1;
        return -1;
    }
};
```
- Time Complexity: O(N*α(N)) where N is the number of connections and α(N) is the Inverse Ackermann function, which is very slow-growing and can be considered almost constant for practical purposes.
- Space Complexity: O(N) for the Disjoint Set Union data structure.
