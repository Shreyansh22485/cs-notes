# Intuition

When merging two trees, the minimum diameter of the resulting tree can be determined by considering the diameters of the individual trees and the path connecting them. The key insight is that the diameter of the merged tree will be the maximum of:
1. The diameter of the first tree.
2. The diameter of the second tree.
3. The path connecting the two trees, which is the sum of the distances from each tree's center to the connecting node plus one (for the edge connecting the trees).



```cpp
class Solution {
public:
    // if we chose any random node on graph then find the node that is furthest , then that will be either end of diameter
    int findDiameter(vector<int> adj[],int n){
        auto [farthestNode,_] = findFarthestNode(adj,0,n);
        auto [_,diameter] = findFarthestNode(adj,farthestNode,n);

        return diameter;
    }
    pair<int,int> findFarthestNode(vector<int> adj[], int sourceNode, int n){
        queue<int> q;
        vector<bool> visited(n,false);
        int distance = 0;
        int farthestNode = sourceNode;

        q.push(sourceNode);
        visited[sourceNode] = true;

        while(!q.empty()){
            int size = q.size();
            while(size--){
                int node = q.front();
                q.pop();
                farthestNode = node;
                for(auto adjNode : adj[node]){
                    if(!visited[adjNode]){
                        visited[adjNode] = true;
                        q.push(adjNode);
                    }
                }
            }

            if(!q.empty()){
                distance++;
            }
        }
        return {farthestNode,distance};

    }
    int minimumDiameterAfterMerge(vector<vector<int>>& edges1, vector<vector<int>>& edges2) {
        int n1 = edges1.size()+1;
        int n2 = edges2.size()+1;

        vector<int> adj1[n1];
        for(auto edge:edges1){
            int u = edge[0];
            int v = edge[1];

            adj1[u].push_back(v);
            adj1[v].push_back(u);
        }
        vector<int> adj2[n2];
        for(auto edge:edges2){
            int u = edge[0];
            int v = edge[1];

            adj2[u].push_back(v);
            adj2[v].push_back(u);
        }
        int diameter1 = findDiameter(adj1,n1);
        int diameter2 = findDiameter(adj2,n2);

        return max({diameter1,diameter2,(diameter1+1)/2 + 1 + (diameter2+1)/2});
    }
};
```
