```cpp
class Solution {
public:
    typedef pair<int,int> P;
    int minCostConnectPoints(vector<vector<int>>& points) {
        int n = points.size();
        map<P,bool> inMST;
        priority_queue<pair<int,P>, vector<pair<int,P>>, greater<pair<int,P>>> pq;

        P source = {points[0][0],points[0][1]};
        pq.push({0,source});

        int sum = 0;

        while(!pq.empty()){
            auto [wt, pointInfo] = pq.top();
            pq.pop();

            int x = pointInfo.first;
            int y = pointInfo.second;

            if(inMST[{x,y}] == true) continue;

            sum += wt;
            inMST[{x,y}] = true;

            for(auto& point: points){
                int x1 = point[0];
                int y1 = point[1];

                if(x1 == x && y1 == y) continue;
                if(!inMST[{x1,y1}]){
                    int val = abs(x1-x) + abs(y1-y);
                    pq.push({val,{x1,y1}});
                }
            }
        }

        return sum;
    }
};
```
- Time Complexity: O(N^2 log N), where N is the number of points.
- Space Complexity: O(N^2) for storing the graph and the priority queue.
