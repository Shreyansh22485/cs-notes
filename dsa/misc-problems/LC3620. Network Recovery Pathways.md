```cpp
class Solution {
public:
    bool isPossible(int score, unordered_map<int,vector<pair<int,int>>>& adj, vector<bool>& online, long long k){
       
        priority_queue< pair<long long, int>, vector< pair<long long, int>> , greater< pair<long long, int>>> pq;
        int n = online.size();
        vector<long long> dist(n,LLONG_MAX);
        pq.push({0,0});
        dist[0] = 0;

        while(!pq.empty()){
            auto [cost, node] = pq.top();
            pq.pop();

            if(node == n-1) return true;
            if(dist[node] < cost) continue;

            for(auto& [adjNode, wt] : adj[node]){
                if(online[adjNode] && wt >= score && cost + wt <= k && cost + wt < dist[adjNode]){
                    pq.push({cost+wt , adjNode});
                    dist[adjNode] = cost+wt;
                }
            }
            
        }
        return false;
    }
    int findMaxPathScore(vector<vector<int>>& edges, vector<bool>& online, long long k) {
        unordered_map<int,vector<pair<int,int>>> adj;
        int maxCost = -1;
        for(auto& edge : edges){
            int u = edge[0];
            int v = edge[1];
            int cost = edge[2];

            maxCost = max(maxCost,cost);
            adj[u].push_back({v,cost});
        }

        int l = 0;
        int r = maxCost;
        int ans = -1;
        while(l<=r){
            int mid = r - (r-l)/2;
            
            if(isPossible(mid,adj,online,k)){
                ans = mid;
                l = mid + 1;
            }
            else{
                r = mid - 1;
            }
        }
        return ans;
    }
};
```
- Time Complexity: O(ElogE + Elog(W))
  - E is the number of edges
  - W is the maximum weight of an edge
- Space Complexity: O(E)
  - E is the number of edges
