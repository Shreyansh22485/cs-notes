```cpp
class Solution {
public:
    vector<vector<int>> directions = {{-1,0},{1,0},{0,1},{0,-1}};
    int n;
    int m;

    bool isPossible(int safeFact, vector<vector<int>>& dist){
        if(safeFact > dist[0][0] || safeFact > dist[n-1][m-1]) return false;

        queue<pair<int,int>>q;
        vector<vector<bool>> vis(n,vector<bool>(m,false));

        q.push({0,0});
        vis[0][0] = true;

        while(!q.empty()){
            auto it = q.front();
            q.pop();

            int i = it.first;
            int j = it.second;

            if(i == n-1 && j==m-1) return true;

            for(auto& dir : directions){
                int x = i + dir[0];
                int y = j + dir[1];

                if(x>=0 && x<n && y>=0 && y<m && !vis[x][y] && dist[x][y] >= safeFact){
                    q.push({x,y});
                    vis[x][y] = 1;
                }
            }
        }

        return false;
    }

    int maximumSafenessFactor(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();

        
        queue<pair<int,int>> q;
        vector<vector<int>> dist(n,vector<int>(m,-1));

        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j] == 1){
                    q.push({i,j});
                    dist[i][j] = 0;
                }
            }
        }

        int level = 0;
        while(!q.empty()){
            int size = q.size();
            while(size--){
                auto it = q.front();
                q.pop();

                int i = it.first;
                int j = it.second;

                for(auto& dir: directions){
                    int x = i + dir[0];
                    int y = j + dir[1];

                    if(x >=0 && x < n && y >=0 && y < m && dist[x][y] == -1){
                        q.push({x,y});
                        dist[x][y] = level+1;
                    }
                }
            }
            if(!q.empty())
                level++;
        }

        int l = 0;
        int r = level;
        int ans = -1;
        while(l <= r){
            int mid = r - (r-l)/2;
            
            if(isPossible(mid,dist)){
                ans = mid;
                l = mid + 1;
            }
            else{
                r = mid - 1;
            }
        }
        
        
        return ans;
    }
};
```

---

**Think of it as**: "Can I find a path from top-left to bottom-right such that every cell on the path has a score of at least `X`?" If I can do it for `X`, I can do it for any `X' < X`. This monotonic property allows binary search.

- For a cell to be "safe" enough, we need to know how close it is to a thief. We can precompute the minimum distance from every cell to the nearest thief using a multi-source BFS starting from all thieves simultaneously. Let this be `dist[i][j]`.
- Now, the problem transforms into finding a path from `(0,0)` to `(n-1,m-1)` such that the minimum `dist[i][j]` value along the path is maximized.
- This is a classic "maximin path" problem. We can observe that if a path exists with a minimum safety factor of `X`, a path also exists for any `X' < X`. This monotonicity allows us to use binary search on the answer (the maximum safety factor).
- So, we binary search for the largest `X` such that a path exists from `(0,0)` to `(n-1,m-1)` using only cells `(i,j)` where `dist[i][j] >= X`. For a fixed `X`, we can check the existence of such a path using a simple BFS or DFS on the grid, considering only "valid" cells (where `dist[i][j] >= X`).

## Time Complexity
- Multi-source BFS to compute distances: O(N*M)
- Binary search: O(log(N*M)) iterations
- Inside binary search, BFS takes O(N*M)
- Total: O(N*M*log(N*M))

## Space Complexity
- Distance matrix: O(N*M)
- BFS queues and visited arrays: O(N*M)
- Total: O(N*M)
