```cpp
class Solution {
public:
    char processStr(string s, long long k) {
        long long L = 0;

        // Simulation to Find L

        for(int i=0;i<s.size();i++){
            char ch = s[i];
            if(ch >= 'a' && ch <= 'z'){
                L++;
            }
            else if(ch == '*'){
                if(L>0)
                    L--;
            }
            else if(ch == '#'){
                L *= 2;
            }
        }
        if(k>=L) return '.';

        // Reverse Simulating
        

        for(int i=s.size()-1;i>=0;i--){
            char ch = s[i];
            if(ch >= 'a' && ch <='z'){
                L--;
                if(L==k){
                    return ch;
                }
            }
            else if(ch == '*'){
                L++;
            }
            else if(ch == '#'){
                L = L/2;
                k=k%L;
            }
            else if(ch == '%'){
                k = L - k - 1;
            }
        }

        return '.';
    }
};
```
- Time Complexity: O(n) where n is the length of the string s.
- Space Complexity: O(1) as we are using only a constant amount of space.
